    // The engine will close the connection on Engine.Close().
    // The caller should not call nc.Close() after passing it here.
    func NewWithNATS(nc *nats.Conn, cfg *Config, opts ...Option) (*Engine, error) {

        }
        if cfg == nil {
            return nil, fmt.Errorf("config is nil")
        }
        cfg.SetDefaults()
        if err := cfg.Validate(); err != nil {
            return nil, err
        }

        if err != nil {
            return nil, fmt.Errorf("creating JetStream context: %w", err)
        }
        eng, err := engine.New(nc, js, cfg, convertOpts(opts...)...)
        if err != nil {
            return nil, err
        }
        return &Engine{Engine: eng, ownedNC: nc}, nil

    // NewEmbedded creates an Engine with an embedded NATS server (D-48: owns shutdown).
    // This is the zero-infrastructure deployment mode.
    // The config path can specify NATS.StoreDir for persistent storage.
    func NewEmbedded(cfg *Config, opts ...Option) (*Engine, error) {
        if cfg == nil {
            return nil, fmt.Errorf("config is nil")
        }
        cfg.SetDefaults()
        if err := cfg.Validate(); err != nil {
            return nil, err
        }
        // Start embedded NATS
        storeDir := cfg.NATS.StoreDir
        node, err := embed.StartNode(embed.NodeConfig{
            StoreDir: storeDir,
        })
        if err != nil {
            return nil, fmt.Errorf("starting embedded NATS: %w", err)
        }
        // Connect
        nc, err := nats.Connect(node.ClientURL(), nats.Timeout(10*time.Second))
        if err != nil {
            node.Shutdown()
            return nil, fmt.Errorf("connecting to embedded NATS: %w", err)
        }
        js, err := jetstream.New(nc)
        if err != nil {
            nc.Close()
            node.Shutdown()
            return nil, fmt.Errorf("creating JetStream context: %w", err)
        }
        eng, err := engine.New(nc, js, cfg, convertOpts(opts...)...)
        if err != nil {
            nc.Close()
            node.Shutdown()
            return nil, err
        }
        return &Engine{Engine: eng, ownedNC: nc, embedNode: node}, nil
    }

    **runServer function:**

    D-51 three-mode design:
    - `natsql server --config=config.yaml` (external NATS via config or default URL)
    - `natsql server --config=config.yaml --embedded` (embedded NATS)
    - `natsql server --config=config.yaml --nats-url=nats://host:4222` (explicit external NATS)

    ```go
    func runServer(cmd *cobra.Command) error {
        // 1. Load config file (D-52: config file is primary source)
        cfg, err := natsql.LoadConfig(cfgPath)
        if err != nil {
            return fmt.Errorf("loading config: %w", err)
        }

        // 2. Apply CLI overrides (D-52: CLI flags override matching config fields)
        if embedded {
            cfg.NATS.Embedded = true
        }
        if natsURL != "" {
            cfg.NATS.URL = natsURL
        }
        if storeDir != "" {
            cfg.NATS.StoreDir = storeDir
        }
        if httpPort > 0 {
            cfg.HTTP.Port = httpPort
        }

        // 3. Apply defaults
        cfg.SetDefaults()

        // 4. Create engine based on mode
        var eng *natsql.Engine
        if cfg.NATS.Embedded {
            eng, err = natsql.NewEmbedded(cfg)
        } else {
            nc, cerr := nats.Connect(cfg.NATS.URL, nats.Timeout(10*time.Second))
            if cerr != nil {
                return fmt.Errorf("connecting to NATS at %s: %w", cfg.NATS.URL, cerr)
            }
            eng, err = natsql.NewWithNATS(nc, cfg)
        }
        if err != nil {
            return fmt.Errorf("creating engine: %w", err)
        }

        // 5. Startup banner (per Specific Ideas in CONTEXT.md)
        logger := slog.Default()
        logger.Info("starting natsql server",
            "mode", map[bool]string{true: "embedded", false: "external"}[cfg.NATS.Embedded],
            "nats_url", map[bool]string{true: "embedded", false: cfg.NATS.URL}[cfg.NATS.Embedded],
            "views", len(cfg.Views),
            "http_port", cfg.HTTP.Port,
        )
        for _, v := range cfg.Views {
            logger.Info("configured view", "name", v.Name, "source_stream", v.SourceStream)
        }

        // 6. Start engine
        ctx, cancel := context.WithCancel(context.Background())
        defer cancel()

        if err := eng.Start(ctx); err != nil {
            return fmt.Errorf("starting engine: %w", err)
        }
        logger.Info("engine started")

        // 7. Wait for SIGINT/SIGTERM (graceful shutdown)
        sigCh := make(chan os.Signal, 1)
        signal.Notify(sigCh, syscall.SIGINT, syscall.SIGTERM)
        sig := <-sigCh
        logger.Info("received signal, shutting down", "signal", sig)

        // 8. Graceful shutdown
        if err := eng.Close(); err != nil {
            return fmt.Errorf("error during shutdown: %w", err)
        }
        logger.Info("server shut down gracefully")
        return nil
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| CLI flags → Config | User-provided command-line flags override config file values |
| Embedded NATS → Engine | Embedded server runs in-process; any Go code in the process can access it |
| Library consumer → Engine | Caller provides JetStream context or NATS connection; full access to that domain |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Tampering | CLI --config loading | mitigate | Config file loaded with os.ReadFile; no remote paths, no exec. If file doesn't exist, error is returned before engine starts. |
| T-03-02 | Information Disclosure | Embedded NATS | accept | Embedded NATS binds to 127.0.0.1:random-port by default. No auth. In-process only — same security boundary as the host process. |
| T-03-03 | Denial of Service | Engine.Close() timeout | mitigate | HTTP server shutdown has 5s timeout (already in code). Materializer drain has 10s context timeout. Close() won't block indefinitely. |
| T-03-04 | Elevation of Privilege | Library facade | accept | Library API (New, NewWithNATS, NewEmbedded) provides same access as calling engine.New directly. No privilege boundary crossed. |
