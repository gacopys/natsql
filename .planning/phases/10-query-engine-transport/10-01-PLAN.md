  // Same PK column with multiple different OpEq values → empty result
  pkSeen := make(map[string]any)
  for _, cond := range q.Where {
      if cond.Op != OpEq {
          continue

      if _, isPK := pkConditions[cond.Column]; isPK {

                  // Contradictory PK conditions → empty result (D-02)
